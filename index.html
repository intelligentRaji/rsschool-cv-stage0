function moveZeros(arr) {
   let arrSmall = [];
   for (let i = 0; i < arr.length; i++){
      let val = arr[i];
      if (val === 0){
         arrSmall.push(...arr.splice(i, 1));
         i--;
      }
   }
   return arr.concat(arrSmall);
}
